const ERR_MAX2 = 'Max 2 bridges per pair'
const ERR_EXCEED = 'Island would exceed its number'
const ERR_CROSS = 'Bridges would cross'

  ERR_NON_ADJACENT,
  ERR_MAX2,
  ERR_EXCEED,
  ERR_CROSS,

export function canonicalBridgeKey(from, to) {
  const a = { x: from.x, y: from.y }
  const b = { x: to.x, y: to.y }
  const first = a.x < b.x || (a.x === b.x && a.y <= b.y) ? a : b
  const second = first === a ? b : a
  return `${first.x},${first.y},${second.x},${second.y}`
}

export function parseBridgeKey(key) {
  const [x1, y1, x2, y2] = key.split(',').map(Number)
  return { x1, y1, x2, y2 }
}

function isAxisAligned(from, to) {
  return from.x === to.x || from.y === to.y
}

export function findIslandAt(islands, x, y) {
  return islands.find((i) => i.x === x && i.y === y) || null
}

// Two islands are adjacent if they are aligned and there is no island between them.
export function areAdjacent(islands, from, to) {
  if (!isAxisAligned(from, to)) return false
  const dx = Math.sign(to.x - from.x)
  const dy = Math.sign(to.y - from.y)
  const steps = Math.max(Math.abs(to.x - from.x), Math.abs(to.y - from.y))
  for (let s = 1; s < steps; s++) {
    const ix = from.x + dx * s
    const iy = from.y + dy * s
    if (findIslandAt(islands, ix, iy)) return false
  }
  return true
}

function segmentIntersects(a1, a2, b1, b2) {
  // Only for axis-aligned segments.
  const aVert = a1.x === a2.x
  const bVert = b1.x === b2.x
  if (aVert === bVert) return false

  const vert = aVert ? { p1: a1, p2: a2 } : { p1: b1, p2: b2 }
  const horz = aVert ? { p1: b1, p2: b2 } : { p1: a1, p2: a2 }

  const vx = vert.p1.x
  const hy = horz.p1.y

  const vMinY = Math.min(vert.p1.y, vert.p2.y)
  const vMaxY = Math.max(vert.p1.y, vert.p2.y)
  const hMinX = Math.min(horz.p1.x, horz.p2.x)
  const hMaxX = Math.max(horz.p1.x, horz.p2.x)

  // Proper crossing must be strictly inside both segments (not at endpoints)
  const within = vx > hMinX && vx < hMaxX && hy > vMinY && hy < vMaxY
  return within
}

export function bridgesWouldCross(bridges, newKey) {
  const { x1, y1, x2, y2 } = parseBridgeKey(newKey)
  const n1 = { x: x1, y: y1 }
  const n2 = { x: x2, y: y2 }

  for (const [key, count] of bridges.entries()) {
    if (!count) continue
    if (key === newKey) continue
    const { x1: a1x, y1: a1y, x2: a2x, y2: a2y } = parseBridgeKey(key)
    const a1 = { x: a1x, y: a1y }
    const a2 = { x: a2x, y: a2y }
    if (segmentIntersects(n1, n2, a1, a2)) return true
  }
  return false
}

export function computeCurrentCounts(islands, bridges) {
  const counts = new Map()
  for (const isl of islands) counts.set(isl.id, 0)

  const byCoord = new Map(islands.map((i) => [`${i.x},${i.y}`, i]))

  for (const [key, num] of bridges.entries()) {
    if (!num) continue
    const { x1, y1, x2, y2 } = parseBridgeKey(key)
    const i1 = byCoord.get(`${x1},${y1}`)
    const i2 = byCoord.get(`${x2},${y2}`)
    if (!i1 || !i2) continue
    counts.set(i1.id, (counts.get(i1.id) || 0) + num)
    counts.set(i2.id, (counts.get(i2.id) || 0) + num)
  }
  return counts
}

export function canAddBridge(islands, bridges, from, to, delta) {
  if (!areAdjacent(islands, from, to)) {
    return { valid: false, error: ERR_NON_ADJACENT }
  }

  const key = canonicalBridgeKey(from, to)
  const current = bridges.get(key) || 0
  const next = current + delta

  if (next < 0) return { valid: true, error: null }
  if (next > 2) return { valid: false, error: ERR_MAX2 }

  const counts = computeCurrentCounts(islands, bridges)
  const fromId = islands.find((i) => i.x === from.x && i.y === from.y)?.id
  const toId = islands.find((i) => i.x === to.x && i.y === to.y)?.id
  if (!fromId || !toId) return { valid: false, error: ERR_NON_ADJACENT }

  const fromIsland = islands.find((i) => i.id === fromId)
  const toIsland = islands.find((i) => i.id === toId)

  const fromNext = (counts.get(fromId) || 0) + delta
  const toNext = (counts.get(toId) || 0) + delta

  if (fromNext > fromIsland.value || toNext > toIsland.value) {
    return { valid: false, error: ERR_EXCEED }
  }

  if (delta > 0) {
    if (bridgesWouldCross(bridges, key)) {
      return { valid: false, error: ERR_CROSS }
    }
  }

  return { valid: true, error: null }
}

export function checkConnectivity(islands, bridges) {
  if (islands.length <= 1) return true

  const byCoord = new Map(islands.map((i) => [`${i.x},${i.y}`, i]))
  const adj = new Map(islands.map((i) => [i.id, new Set()]))

  for (const [key, num] of bridges.entries()) {
    if (!num) continue
    const { x1, y1, x2, y2 } = parseBridgeKey(key)
    const i1 = byCoord.get(`${x1},${y1}`)
    const i2 = byCoord.get(`${x2},${y2}`)
    if (!i1 || !i2) continue
    adj.get(i1.id).add(i2.id)
    adj.get(i2.id).add(i1.id)
  }

  const start = islands[0].id
  const seen = new Set([start])
  const q = [start]

  while (q.length) {
    const cur = q.shift()
    for (const nxt of adj.get(cur)) {
      if (!seen.has(nxt)) {
        seen.add(nxt)
        q.push(nxt)
      }
    }
  }

  return seen.size === islands.length
}

export function isVictory(islands, bridges) {
  // Ensure no crossings at all
  for (const [key, num] of bridges.entries()) {
    if (!num) continue
    if (bridgesWouldCross(bridges, key)) return false
  }

  const counts = computeCurrentCounts(islands, bridges)
  for (const isl of islands) {
    if ((counts.get(isl.id) || 0) !== isl.value) return false
  }

  return checkConnectivity(islands, bridges)
}
