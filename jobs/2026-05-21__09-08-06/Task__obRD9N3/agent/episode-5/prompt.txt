Current Terminal Screen:
          {islands.map((isl) => {
            const px = isl.x * GRID;
            const py = isl.y * GRID;
            const isSelected = selectedId === isl.id || kbdFromId === isl.id;
            const current = counts.get(isl.id) ?? 0;
            const satisfied = current === isl.value;

            return (
              <g key={isl.id}>
                <circle
                  cx={px}
                  cy={py}
                  r={16}
                  fill="#000"
                  stroke={isSelected ? '#00ff66' : 'rgba(0,255,102,0.7)'}
                  strokeWidth={isSelected ? 4 : 2}
                  onClick={(e) => {
                    onIslandAuxOrCtrlClick(isl, e);
                    if (!isCtrlLike(e)) onIslandClick(isl, e);
                  }}
                  onContextMenu={(e) => onIslandContextMenu(isl, e)}
                />
                <text
                  x={px}
                  y={py + 6}
                  textAnchor="middle"
                  fontSize="16"
                  fill={satisfied ? '#00ff66' : 'rgba(0,255,102,0.9)'}
                >
                  {isl.value}
                </text>
              </g>
            );
          })}
        </g>
      </svg>
    </div>
  );
