export function canonicalKey(from, to) {
  const a = { x: from.x, y: from.y };
  const b = { x: to.x, y: to.y };
  const first = (a.x < b.x) || (a.x === b.x && a.y <= b.y) ? a : b;
  const second = first === a ? b : a;
  return `${first.x},${first.y},${second.x},${second.y}`;
}

export function parseKey(key) {
  const [x1, y1, x2, y2] = key.split(',').map(Number);
  return { x1, y1, x2, y2 };
}

export function isStraight(from, to) {
  return from.x === to.x || from.y === to.y;
}

export function hasIslandBetween(islands, from, to) {
  if (!isStraight(from, to)) return true;
  if (from.x === to.x) {
    const x = from.x;
    const yMin = Math.min(from.y, to.y);
    const yMax = Math.max(from.y, to.y);
    return islands.some((isl) => isl.x === x && isl.y > yMin && isl.y < yMax);
  }
  const y = from.y;
  const xMin = Math.min(from.x, to.x);
  const xMax = Math.max(from.x, to.x);
  return islands.some((isl) => isl.y === y && isl.x > xMin && isl.x < xMax);
}

export function computeCurrentCounts(islands, bridges) {
  const counts = new Map();
  for (const isl of islands) counts.set(isl.id, 0);

  for (const [key, n] of bridges.entries()) {
    if (!n) continue;
    const { x1, y1, x2, y2 } = parseKey(key);
    const a = islands.find((i) => i.x === x1 && i.y === y1);
    const b = islands.find((i) => i.x === x2 && i.y === y2);
    if (!a || !b) continue;
    counts.set(a.id, (counts.get(a.id) ?? 0) + n);
    counts.set(b.id, (counts.get(b.id) ?? 0) + n);
  }
  return counts;
}

function segmentsCross(segA, segB) {
  // seg: {x1,y1,x2,y2} in grid units, axis-aligned, endpoints at islands.
  const aVert = segA.x1 === segA.x2;
  const bVert = segB.x1 === segB.x2;
  if (aVert === bVert) return false; // parallel can't cross in interior in Hashi sense.

  const v = aVert ? segA : segB;
  const h = aVert ? segB : segA;

  const vx = v.x1;
  const vyMin = Math.min(v.y1, v.y2);
  const vyMax = Math.max(v.y1, v.y2);

  const hy = h.y1;
  const hxMin = Math.min(h.x1, h.x2);
  const hxMax = Math.max(h.x1, h.x2);

  const ix = vx;
  const iy = hy;

  // intersection must be strictly inside both segments, not at endpoints.
  const vInterior = iy > vyMin && iy < vyMax;
  const hInterior = ix > hxMin && ix < hxMax;
  return vInterior && hInterior;
}

export function bridgesWouldCross(bridges, from, to) {
  const candidate = { x1: from.x, y1: from.y, x2: to.x, y2: to.y };

    const same = (seg.x1 === candidate.x1 && seg.y1 === candidate.y1 && seg.x2 === candidate.x2 && seg.y2 === candidate.y2) ||

}

export function canAddBridge(islands, bridges, from, to, delta) {
  if (!from || !to || from.id === to.id) return { valid: false, error: ERR_NON_ADJ };
  if (!isStraight(from, to)) return { valid: false, error: ERR_NON_ADJ };
  if (hasIslandBetween(islands, from, to)) return { valid: false, error: ERR_NON_ADJ };

  const key = canonicalKey(from, to);
  const existing = bridges.get(key) ?? 0;
  const next = existing + delta;
  if (next < 0) return { valid: true, error: null };
  if (next > 2) return { valid: false, error: ERR_MAX2 };

  const counts = computeCurrentCounts(islands, bridges);
  const fromCount = (counts.get(from.id) ?? 0) + delta;
  const toCount = (counts.get(to.id) ?? 0) + delta;
  if (fromCount > from.value || toCount > to.value) return { valid: false, error: ERR_EXCEED };

  // crossing only matters if we're adding (delta>0)
  if (delta > 0 && bridgesWouldCross(bridges, from, to)) return { valid: false, error: ERR_CROSS };

  return { valid: true, error: null };
}
