const ERR_NON_ADJACENT = 'Cannot connect non-adjacent islands'
const ERR_MAX_PAIR = 'Max 2 bridges per pair'
const ERR_EXCEED = 'Island would exceed its number'
const ERR_CROSS = 'Bridges would cross'

export function canonicalBridgeKey(from, to) {
  const a = { x: from.x, y: from.y }
  const b = { x: to.x, y: to.y }
  if (a.x < b.x || (a.x === b.x && a.y <= b.y)) {
    return `${a.x},${a.y},${b.x},${b.y}`
  }
  return `${b.x},${b.y},${a.x},${a.y}`
}

function parseKey(key) {
  const [x1, y1, x2, y2] = key.split(',').map(Number)
  return { x1, y1, x2, y2 }
}

function isAligned(a, b) {
  return a.x === b.x || a.y === b.y
}

function between(v, a, b) {
  return v > Math.min(a, b) && v < Math.max(a, b)
}

export function getIslandAt(islands, x, y) {
  return islands.find((i) => i.x === x && i.y === y) || null
}

export function areAdjacent(islands, from, to) {
  if (!from || !to) return false
  if (!isAligned(from, to)) return false
  if (from.x === to.x) {
    const x = from.x
    const yMin = Math.min(from.y, to.y)
    const yMax = Math.max(from.y, to.y)
    for (const i of islands) {
      if (i.id === from.id || i.id === to.id) continue
      if (i.x === x && i.y > yMin && i.y < yMax) return false
    }
    return true
  }
  const y = from.y
  const xMin = Math.min(from.x, to.x)
  const xMax = Math.max(from.x, to.x)
  for (const i of islands) {
    if (i.id === from.id || i.id === to.id) continue
    if (i.y === y && i.x > xMin && i.x < xMax) return false
  }
  return true
}

export function computeCurrentCounts(islands, bridges) {
  const counts = new Map()
  for (const isl of islands) counts.set(isl.id, 0)

  for (const [key, n] of bridges.entries()) {
    if (!n) continue
    const { x1, y1, x2, y2 } = parseKey(key)
    const a = getIslandAt(islands, x1, y1)
    const b = getIslandAt(islands, x2, y2)
    if (!a || !b) continue
    counts.set(a.id, (counts.get(a.id) ?? 0) + n)
    counts.set(b.id, (counts.get(b.id) ?? 0) + n)
  }
  return counts
}

function segmentsCross(k1, k2) {
  const s1 = parseKey(k1)
  const s2 = parseKey(k2)

  const s1Vert = s1.x1 === s1.x2
  const s2Vert = s2.x1 === s2.x2
  if (s1Vert === s2Vert) return false

  const v = s1Vert ? s1 : s2
  const h = s1Vert ? s2 : s1

  const ix = v.x1
  const iy = h.y1

  // Proper crossing (not at endpoints)
  const vYMin = Math.min(v.y1, v.y2)
  const vYMax = Math.max(v.y1, v.y2)
  const hXMin = Math.min(h.x1, h.x2)
  const hXMax = Math.max(h.x1, h.x2)

  if (!between(iy, vYMin, vYMax)) return false
  if (!between(ix, hXMin, hXMax)) return false
  return true
}

function wouldCrossExisting(bridges, candidateKey) {
  for (const [k, n] of bridges.entries()) {
    if (!n) continue
    if (k === candidateKey) continue
    if (segmentsCross(k, candidateKey)) return true
  }
  return false
}

export function canAddBridge(islands, bridges, from, to, delta) {
  if (!from || !to) return { valid: false, error: ERR_NON_ADJACENT }
  if (from.id === to.id) return { valid: false, error: ERR_NON_ADJACENT }
  if (!areAdjacent(islands, from, to)) return { valid: false, error: ERR_NON_ADJACENT }

  const key = canonicalBridgeKey(from, to)
  const existing = bridges.get(key) ?? 0
  const next = existing + delta

  if (next < 0) return { valid: true, error: null }
  if (next > 2) return { valid: false, error: ERR_MAX_PAIR }

  const counts = computeCurrentCounts(islands, bridges)
  const fromCount = counts.get(from.id) ?? 0
  const toCount = counts.get(to.id) ?? 0

  // Only enforce exceed when adding
  if (delta > 0) {
    if (fromCount + delta > from.value || toCount + delta > to.value) {
      return { valid: false, error: ERR_EXCEED }
    }
  }

  // Crossing only matters if we will have at least 1 bridge after change
  if (delta > 0 && next > 0) {
    const tmp = new Map(bridges)
    tmp.set(key, next)
    // candidate segment is the same regardless of single/double
    if (wouldCrossExisting(bridges, key)) {
      return { valid: false, error: ERR_CROSS }
    }
  }

  return { valid: true, error: null }
}

export function checkConnectivity(islands, bridges) {
  if (islands.length === 0) return true
  const adj = new Map()
  for (const i of islands) adj.set(i.id, [])

  for (const [key, n] of bridges.entries()) {
    if (!n) continue
    const { x1, y1, x2, y2 } = parseKey(key)
    const a = getIslandAt(islands, x1, y1)
    const b = getIslandAt(islands, x2, y2)
    if (!a || !b) continue
    adj.get(a.id).push(b.id)
    adj.get(b.id).push(a.id)
  }

  const startId = islands[0].id
  const q = [startId]
  const seen = new Set([startId])
  while (q.length) {
    const cur = q.shift()
    for (const nxt of adj.get(cur) || []) {
      if (!seen.has(nxt)) {
        seen.add(nxt)
        q.push(nxt)
      }
    }
  }
  return seen.size === islands.length
}

export function anyCrossing(bridges) {
  const keys = [...bridges.entries()].filter(([, n]) => n > 0).map(([k]) => k)
  for (let i = 0; i < keys.length; i++) {
    for (let j = i + 1; j < keys.length; j++) {
      if (segmentsCross(keys[i], keys[j])) return true
    }
  }
  return false
}

export function isVictory(islands, bridges) {
  const counts = computeCurrentCounts(islands, bridges)
  for (const isl of islands) {
    if ((counts.get(isl.id) ?? 0) !== isl.value) return false
  }
  if (anyCrossing(bridges)) return false
  if (!checkConnectivity(islands, bridges)) return false
  return true
}
