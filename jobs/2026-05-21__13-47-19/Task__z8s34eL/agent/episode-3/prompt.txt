export function canonicalBridgeKey(from, to) {
  // from/to are {x,y} in grid units
  const a = { x: from.x, y: from.y };
  const b = { x: to.x, y: to.y };
  const swap = a.x > b.x || (a.x === b.x && a.y > b.y);
  const p = swap ? b : a;
  const q = swap ? a : b;
  return `${p.x},${p.y},${q.x},${q.y}`;
}

function parseKey(key) {
  const [x1, y1, x2, y2] = key.split(',').map(Number);
  return { x1, y1, x2, y2 };
}

export function computeCurrentCounts(islands, bridges) {
  const counts = new Map();
  for (const isl of islands) counts.set(isl.id, 0);

  for (const [key, n] of bridges.entries()) {
    if (!n) continue;
    const { x1, y1, x2, y2 } = parseKey(key);
    const a = islands.find((i) => i.x === x1 && i.y === y1);
    const b = islands.find((i) => i.x === x2 && i.y === y2);
    if (!a || !b) continue;
    counts.set(a.id, (counts.get(a.id) ?? 0) + n);
    counts.set(b.id, (counts.get(b.id) ?? 0) + n);
  }

  return counts;
}

function isCardinalAdjacent(islands, from, to) {
  if (from.id === to.id) return { ok: false, reason: 'same' };
  if (from.x !== to.x && from.y !== to.y) return { ok: false, reason: 'not_cardinal' };

  if (from.x === to.x) {
    const x = from.x;
    const minY = Math.min(from.y, to.y);
    const maxY = Math.max(from.y, to.y);
    // any island strictly between blocks adjacency
    const between = islands.some((i) => i.x === x && i.y > minY && i.y < maxY);
    return { ok: !between, reason: between ? 'blocked' : null };
  }

  const y = from.y;
  const minX = Math.min(from.x, to.x);
  const maxX = Math.max(from.x, to.x);
  const between = islands.some((i) => i.y === y && i.x > minX && i.x < maxX);
  return { ok: !between, reason: between ? 'blocked' : null };
}

function bridgeSegmentsFromKey(key) {
  const { x1, y1, x2, y2 } = parseKey(key);
  const horiz = y1 === y2;
  const vert = x1 === x2;
  if (!horiz && !vert) return null;
  if (horiz) {
    const y = y1;
    const minX = Math.min(x1, x2);
    const maxX = Math.max(x1, x2);
    return { type: 'h', y, x1: minX, x2: maxX };
  }
  const x = x1;
  const minY = Math.min(y1, y2);
  const maxY = Math.max(y1, y2);
  return { type: 'v', x, y1: minY, y2: maxY };
}

function segmentsCross(segA, segB) {
  // Bridges that share an endpoint (i.e. touch at an island) are OK.
  if (!segA || !segB) return false;
  if (segA.type === segB.type) return false; // parallel won't cross in Hashi model

  const h = segA.type === 'h' ? segA : segB;
  const v = segA.type === 'v' ? segA : segB;

  // intersection point
  const ix = v.x;
  const iy = h.y;

  const withinH = ix > h.x1 && ix < h.x2;
  const withinV = iy > v.y1 && iy < v.y2;
  return withinH && withinV;
}

function wouldCrossExisting(bridges, newKey) {
  const newSeg = bridgeSegmentsFromKey(newKey);
  for (const [key, n] of bridges.entries()) {
    if (!n) continue;
    if (key === newKey) continue;
    const seg = bridgeSegmentsFromKey(key);
    if (segmentsCross(newSeg, seg)) return true;
  }
  return false;
}
