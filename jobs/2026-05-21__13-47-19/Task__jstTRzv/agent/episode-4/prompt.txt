function noCrossings(bridges) {
  const segs = [];
  for (const [k, n] of bridges.entries()) {
    if (!n) continue;
    segs.push(normalizedSeg(parseKey(k)));
  }
  for (let i = 0; i < segs.length; i++) {
    for (let j = i + 1; j < segs.length; j++) {
      if (segmentsCross(segs[i], segs[j])) return false;
    }
  }
  return true;
}

export function isVictory(islands, bridges) {
  const counts = computeCurrentCounts(islands, bridges);
  for (const isl of islands) {
    if ((counts.get(isl.id) ?? 0) !== isl.value) return false;
  }
  if (!noCrossings(bridges)) return false;
  if (!checkConnectivity(islands, bridges)) return false;
  return true;
}
