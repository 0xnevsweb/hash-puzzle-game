const H = 600;

function islandCenterPx(isl) {
  return { cx: isl.x * GRID, cy: isl.y * GRID };
}

function isCtrlClick(e) {
  return e.ctrlKey || e.metaKey;
}

  function updateBridge(from, to, nextValue) {
    const key = getCanonicalBridgeKey(from, to);
    setBridges((prev) => {
      const m = new Map(prev);
      if (!nextValue) m.delete(key);
      else m.set(key, nextValue);
      return m;
    });

  function handleToggle(from, to) {
    const key = getCanonicalBridgeKey(from, to);
    const current = bridges.get(key) ?? 0;
    const next = (current + 1) % 3; // 0->1->2->0
    if (next === 0) {
      updateBridge(from, to, 0);
      clearErrorOnValidMove();
      return;
    }

    const delta = next - current;
    const { valid, error: err } = canAddBridge(islands, bridges, from, to, delta);
    if (!valid) {
      setError(err);
      return;
    }

    updateBridge(from, to, next);
    clearErrorOnValidMove();
  }

  function handleRemoveBridge(from, to) {
    const key = getCanonicalBridgeKey(from, to);
    if ((bridges.get(key) ?? 0) === 0) return;
    updateBridge(from, to, 0);
    clearErrorOnValidMove();
  }

  function onIslandLeftClick(isl) {
    if (selectedId === null) {
      setSelectedId(isl.id);
      return;

    if (selectedId === isl.id) {
      // keep selected; no toggle
      return;

  function moveCursor(dir) {
    if (cursorId === null) return;
    const cur = islands.find((i) => i.id === cursorId);
    if (!cur) return;

    const candidates = getAdjacentIslands(islands, cur);
    // choose based on direction by comparing coordinates
    let next = null;
    for (const c of candidates) {
      if (dir === 'ArrowLeft' && c.y === cur.y && c.x < cur.x) next = c;
      if (dir === 'ArrowRight' && c.y === cur.y && c.x > cur.x) next = c;
