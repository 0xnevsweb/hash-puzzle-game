function isAligned(a, b) {
  return a.x === b.x || a.y === b.y;
}

function canonicalKey(from, to) {
  const a = { x: from.x, y: from.y };
  const b = { x: to.x, y: to.y };
  if (a.x < b.x) return `${a.x},${a.y},${b.x},${b.y}`;
  if (a.x > b.x) return `${b.x},${b.y},${a.x},${a.y}`;
  if (a.y <= b.y) return `${a.x},${a.y},${b.x},${b.y}`;
  return `${b.x},${b.y},${a.x},${a.y}`;
}

function parseKey(key) {
  const [x1, y1, x2, y2] = key.split(',').map(Number);
  return { x1, y1, x2, y2 };
}

function islandAt(islands, x, y) {
  return islands.find((i) => i.x === x && i.y === y) || null;
}

function isAdjacent(islands, from, to) {
  if (!from || !to) return false;
  if (from.id === to.id) return false;
  if (!isAligned(from, to)) return false;

  const dx = Math.sign(to.x - from.x);
  const dy = Math.sign(to.y - from.y);
  let cx = from.x + dx;
  let cy = from.y + dy;
  while (cx !== to.x || cy !== to.y) {
    if (islandAt(islands, cx, cy)) return false;
    cx += dx;
    cy += dy;
  }
  return true;
}

function segmentsCross(segA, segB) {
  // seg: {x1,y1,x2,y2} in grid units, axis-aligned
  const a = { ...segA };
  const b = { ...segB };
  const aH = a.y1 === a.y2;
  const bH = b.y1 === b.y2;
  if (aH === bH) return false; // parallel can't cross in interior (overlap not possible in valid game)

  const h = aH ? a : b;
  const v = aH ? b : a;

  const hxMin = Math.min(h.x1, h.x2);
  const hxMax = Math.max(h.x1, h.x2);
  const vyMin = Math.min(v.y1, v.y2);
  const vyMax = Math.max(v.y1, v.y2);

  const ix = v.x1;
  const iy = h.y1;

  // interior intersection only (not at endpoints)
  const withinH = ix > hxMin && ix < hxMax;
  const withinV = iy > vyMin && iy < vyMax;
  if (!withinH || !withinV) return false;

  return true;
}

function wouldCrossExisting(bridges, from, to) {
  const newSeg = { x1: from.x, y1: from.y, x2: to.x, y2: to.y };
  for (const [key, count] of bridges.entries()) {
    if (!count) continue;
    const seg = parseKey(key);
    // share endpoints => allowed
    const sharesEndpoint =
      (seg.x1 === newSeg.x1 && seg.y1 === newSeg.y1) ||
      (seg.x2 === newSeg.x1 && seg.y2 === newSeg.y1) ||
      (seg.x1 === newSeg.x2 && seg.y1 === newSeg.y2) ||
      (seg.x2 === newSeg.x2 && seg.y2 === newSeg.y2);
    if (sharesEndpoint) continue;

    if (segmentsCross(seg, newSeg)) return true;
  }
  return false;
}

export function computeCurrentCounts(islands, bridges) {
  const counts = new Map();

  for (const [key, n] of bridges.entries()) {

}

export function canAddBridge(islands, bridges, from, to, delta) {
