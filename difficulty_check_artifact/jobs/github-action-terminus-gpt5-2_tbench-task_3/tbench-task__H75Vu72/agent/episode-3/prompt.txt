const ERR_NON_ADJ = 'Cannot connect non-adjacent islands';
const ERR_MAX_PAIR = 'Max 2 bridges per pair';
const ERR_EXCEED = 'Island would exceed its number';
const ERR_CROSS = 'Bridges would cross';

export function canonicalKey(from, to) {
  const a = { x: from.x, y: from.y };
  const b = { x: to.x, y: to.y };
  const first = a.x < b.x || (a.x === b.x && a.y <= b.y) ? a : b;
  const second = first === a ? b : a;
  return `${first.x},${first.y},${second.x},${second.y}`;
}

export function parseKey(key) {
  const [x1, y1, x2, y2] = key.split(',').map((n) => Number(n));
  return { x1, y1, x2, y2 };
}

export function isAxisAligned(a, b) {
  return a.x === b.x || a.y === b.y;
}

export function segmentFromIslands(a, b) {
  const horiz = a.y === b.y;
  if (horiz) {
    const x1 = Math.min(a.x, b.x);
    const x2 = Math.max(a.x, b.x);
    return { orient: 'h', x1, x2, y: a.y };
  }
  const y1 = Math.min(a.y, b.y);
  const y2 = Math.max(a.y, b.y);
  return { orient: 'v', y1, y2, x: a.x };
}

export function areAdjacent(islands, a, b) {
  if (!isAxisAligned(a, b)) return false;
  const seg = segmentFromIslands(a, b);
  if (seg.orient === 'h') {
    const minX = seg.x1;
    const maxX = seg.x2;
    return !islands.some((i) => i.y === seg.y && i.x > minX && i.x < maxX);
  }
  const minY = seg.y1;
  const maxY = seg.y2;
  return !islands.some((i) => i.x === seg.x && i.y > minY && i.y < maxY);
}

export function segmentsCross(segA, segB) {
  if (segA.orient === segB.orient) return false;
  const h = segA.orient === 'h' ? segA : segB;
  const v = segA.orient === 'v' ? segA : segB;

  const withinX = v.x > h.x1 && v.x < h.x2;
  const withinY = h.y > v.y1 && h.y < v.y2;
  return withinX && withinY;
}

export function computeCurrentCounts(islands, bridges) {
  const counts = new Map();
  for (const isl of islands) counts.set(isl.id, 0);

  for (const [key, n] of bridges.entries()) {
    if (!n) continue;
    const { x1, y1, x2, y2 } = parseKey(key);
    const a = islands.find((i) => i.x === x1 && i.y === y1);
    const b = islands.find((i) => i.x === x2 && i.y === y2);
    if (!a || !b) continue;
    counts.set(a.id, (counts.get(a.id) ?? 0) + n);
    counts.set(b.id, (counts.get(b.id) ?? 0) + n);
  }
  return counts;
}

export function wouldCrossExisting(islands, bridges, from, to) {

  for (const [key, n] of bridges.entries()) {

      (from.x === x1 && from.y === y1) ||
      (from.x === x2 && from.y === y2) ||
      (to.x === x1 && to.y === y1) ||
      (to.x === x2 && to.y === y2);
    if (sharesEndpoint) continue;

    const a = islands.find((i) => i.x === x1 && i.y === y1);
    const b = islands.find((i) => i.x === x2 && i.y === y2);
    if (!a || !b) continue;

    const segOld = segmentFromIslands(a, b);
    if (segmentsCross(segNew, segOld)) return true;

  return false;
}

export function canAddBridge(islands, bridges, from, to, delta) {
  if (!from || !to) return { valid: false, error: ERR_NON_ADJ };
  if (from.id === to.id) return { valid: false, error: ERR_NON_ADJ };
  if (!isAxisAligned(from, to) || !areAdjacent(islands, from, to)) {
    return { valid: false, error: ERR_NON_ADJ };
  }

  const key = canonicalKey(from, to);
  const current = bridges.get(key) ?? 0;
  const next = current + delta;

  if (next < 0) return { valid: true, error: null };
  if (next > 2) return { valid: false, error: ERR_MAX_PAIR };

  const counts = computeCurrentCounts(islands, bridges);
  const fromCount = (counts.get(from.id) ?? 0) + delta;
  const toCount = (counts.get(to.id) ?? 0) + delta;
  if (fromCount > from.value || toCount > to.value) {
    return { valid: false, error: ERR_EXCEED };
  }

  if (delta > 0 && wouldCrossExisting(islands, bridges, from, to)) {
    return { valid: false, error: ERR_CROSS };
  }

  return { valid: true, error: null };
}
