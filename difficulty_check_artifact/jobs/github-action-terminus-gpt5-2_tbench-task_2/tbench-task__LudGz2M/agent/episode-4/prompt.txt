export const ERROR_NON_ADJACENT = 'Cannot connect non-adjacent islands';
export const ERROR_MAX_PAIR = 'Max 2 bridges per pair';
export const ERROR_EXCEED = 'Island would exceed its number';

const isBetweenExclusive = (v, a, b) => {
  const min = Math.min(a, b);
  const max = Math.max(a, b);
  return v > min && v < max;
};

export function canonicalBridgeKey(from, to) {
  const a = { x: from.x, y: from.y };
  const b = { x: to.x, y: to.y };
  const first = a.x < b.x || (a.x === b.x && a.y <= b.y) ? a : b;
  const second = first === a ? b : a;
  return `${first.x},${first.y},${second.x},${second.y}`;
}

export function parseBridgeKey(key) {
  const [x1, y1, x2, y2] = key.split(',').map(Number);
  return { x1, y1, x2, y2 };
}

export function computeCurrentCounts(islands, bridges) {
  const counts = new Map();
  for (const isl of islands) counts.set(isl.id, 0);
  for (const [key, n] of bridges.entries()) {
    if (!n) continue;
    const { x1, y1, x2, y2 } = parseBridgeKey(key);
    const a = islands.find((i) => i.x === x1 && i.y === y1);
    const b = islands.find((i) => i.x === x2 && i.y === y2);
    if (!a || !b) continue;
    counts.set(a.id, (counts.get(a.id) ?? 0) + n);
    counts.set(b.id, (counts.get(b.id) ?? 0) + n);
  }
  return counts;
}

function isAdjacent(islands, from, to) {
  if (from.id === to.id) return false;
  const sameRow = from.y === to.y;
  const sameCol = from.x === to.x;
  if (!sameRow && !sameCol) return false;

  // Ensure no island blocks between them.
  if (sameRow) {
    for (const isl of islands) {
      if (isl.id === from.id || isl.id === to.id) continue;
      if (isl.y === from.y && isBetweenExclusive(isl.x, from.x, to.x)) return false;
    }
  } else {
    for (const isl of islands) {
      if (isl.id === from.id || isl.id === to.id) continue;
      if (isl.x === from.x && isBetweenExclusive(isl.y, from.y, to.y)) return false;
    }
  }
  return true;
}

function segmentsCross(segA, segB) {
  // Only consider perpendicular crossing (Hashi rule). Shared endpoints are allowed.
  const aVert = segA.x1 === segA.x2;
  const bVert = segB.x1 === segB.x2;
  if (aVert === bVert) return false;

  const v = aVert ? segA : segB;
  const h = aVert ? segB : segA;

  // Intersection point (v.x, h.y)
  const ix = v.x1;
  const iy = h.y1;

  // Must be strictly inside both segments (not at endpoints)
  if (!isBetweenExclusive(ix, h.x1, h.x2)) return false;
  if (!isBetweenExclusive(iy, v.y1, v.y2)) return false;
  return true;
}

function wouldCrossExisting(bridges, newSeg) {
  for (const [key, n] of bridges.entries()) {
    if (!n) continue;
    const seg = parseBridgeKey(key);
    // Sharing endpoints is fine.
    const sharesEndpoint =
      (seg.x1 === newSeg.x1 && seg.y1 === newSeg.y1) ||
      (seg.x2 === newSeg.x1 && seg.y2 === newSeg.y1) ||
      (seg.x1 === newSeg.x2 && seg.y1 === newSeg.y2) ||
      (seg.x2 === newSeg.x2 && seg.y2 === newSeg.y2);
    if (sharesEndpoint) continue;

    if (segmentsCross(seg, newSeg)) return true;
  }
  return false;
}

export function canAddBridge(islands, bridges, from, to, delta) {
  if (!isAdjacent(islands, from, to)) return { valid: false, error: ERROR_NON_ADJACENT };

  const key = canonicalBridgeKey(from, to);
  const existing = bridges.get(key) ?? 0;
  const next = existing + delta;

  if (next < 0) return { valid: false, error: ERROR_MAX_PAIR };
  if (next > 2) return { valid: false, error: ERROR_MAX_PAIR };

  const counts = computeCurrentCounts(islands, bridges);
  const fromCount = counts.get(from.id) ?? 0;
  const toCount = counts.get(to.id) ?? 0;

  if (fromCount + delta > from.value) return { valid: false, error: ERROR_EXCEED };
  if (toCount + delta > to.value) return { valid: false, error: ERROR_EXCEED };

  if (delta > 0 && existing === 0) {
    const seg = parseBridgeKey(key);
    if (wouldCrossExisting(bridges, seg)) return { valid: false, error: ERROR_CROSS };
  }

  return { valid: true, error: null };
}
