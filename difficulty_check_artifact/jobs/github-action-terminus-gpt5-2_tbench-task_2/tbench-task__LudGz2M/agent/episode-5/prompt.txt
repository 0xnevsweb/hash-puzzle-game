export function checkConnectivity(islands, bridges) {
  if (islands.length === 0) return true;
  const byCoord = new Map();
  for (const isl of islands) byCoord.set(`${isl.x},${isl.y}`, isl);

  const adj = new Map();
  for (const isl of islands) adj.set(isl.id, []);

  for (const [key, n] of bridges.entries()) {
    if (!n) continue;
    const { x1, y1, x2, y2 } = parseBridgeKey(key);
    const a = byCoord.get(`${x1},${y1}`);
    const b = byCoord.get(`${x2},${y2}`);
    if (!a || !b) continue;
    adj.get(a.id).push(b.id);
    adj.get(b.id).push(a.id);
  }

  const startId = islands[0].id;
  const seen = new Set([startId]);
  const stack = [startId];
  while (stack.length) {
    const cur = stack.pop();
    for (const nxt of adj.get(cur) || []) {
      if (seen.has(nxt)) continue;
      seen.add(nxt);
      stack.push(nxt);
    }
  }
  return seen.size === islands.length;
}

export function isVictory(islands, bridges) {
  // Ensure no crossings
  for (const [k1, n1] of bridges.entries()) {
    if (!n1) continue;
    const s1 = parseBridgeKey(k1);
    for (const [k2, n2] of bridges.entries()) {
      if (!n2) continue;
      if (k1 === k2) continue;
      const s2 = parseBridgeKey(k2);
      const sharesEndpoint =
        (s1.x1 === s2.x1 && s1.y1 === s2.y1) ||
        (s1.x1 === s2.x2 && s1.y1 === s2.y2) ||
        (s1.x2 === s2.x1 && s1.y2 === s2.y1) ||
        (s1.x2 === s2.x2 && s1.y2 === s2.y2);
      if (sharesEndpoint) continue;
      if (segmentsCross(s1, s2)) return false;
    }
  }

  const counts = computeCurrentCounts(islands, bridges);
  for (const isl of islands) {
    if ((counts.get(isl.id) ?? 0) !== isl.value) return false;
  }

  return checkConnectivity(islands, bridges);
}
