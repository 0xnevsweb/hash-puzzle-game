function islandsInDirection(islands, from, dir) {
  // Return nearest island strictly in direction if exists.
  // dir: 'left'|'right'|'up'|'down'
  let best = null;
  for (const isl of islands) {
    if (isl.id === from.id) continue;
    if (dir === 'left' && isl.y === from.y && isl.x < from.x) {
      if (!best || isl.x > best.x) best = isl;
    }
    if (dir === 'right' && isl.y === from.y && isl.x > from.x) {
      if (!best || isl.x < best.x) best = isl;
    }
    if (dir === 'up' && isl.x === from.x && isl.y < from.y) {
      if (!best || isl.y > best.y) best = isl;
    }
    if (dir === 'down' && isl.x === from.x && isl.y > from.y) {
      if (!best || isl.y < best.y) best = isl;
    }
  }
  return best;
}

function bridgeLinesForKey(key, count) {

export function computeCurrentCounts(islands, bridges) {
  const counts = new Map();
  for (const isl of islands) counts.set(isl.id, 0);

  for (const [key, count] of bridges.entries()) {
    if (!count) continue;
    const { x1, y1, x2, y2 } = parseBridgeKey(key);
    const a = islands.find((i) => i.x === x1 && i.y === y1);
    const b = islands.find((i) => i.x === x2 && i.y === y2);
    if (!a || !b) continue;
    counts.set(a.id, (counts.get(a.id) ?? 0) + count);
    counts.set(b.id, (counts.get(b.id) ?? 0) + count);
  }
  return counts;
}

export function canAddBridge(islands, bridges, from, to, delta) {
  if (!from || !to) return { valid: false, error: 'Invalid islands' };
  if (from.id === to.id) return { valid: false, error: 'Cannot connect island to itself' };
  if (!isAxisAligned(from, to) || !areAdjacent(islands, from, to)) {
    return { valid: false, error: 'Cannot connect non-adjacent islands' };
  }

  const key = canonicalBridgeKey(from, to);
  const current = bridges.get(key) ?? 0;
  const next = current + delta;

  if (next > 2) return { valid: false, error: 'Max 2 bridges per pair' };
  if (next < 0) return { valid: false, error: 'Invalid bridge count' };

  // Crossing check is for adding (delta > 0) only
  if (delta > 0) {
    const temp = new Map(bridges);
    temp.set(key, next);
    if (bridgesWouldCross(islands, temp, from, to)) {
      return { valid: false, error: 'Bridges would cross' };
    }
  }

  // Island capacity check
  const counts = computeCurrentCounts(islands, bridges);
  const fromCount = counts.get(from.id) ?? 0;
  const toCount = counts.get(to.id) ?? 0;

  if (delta > 0) {
    if (fromCount + delta > from.value || toCount + delta > to.value) {
      return { valid: false, error: 'Island would exceed its number' };
    }
  }

  return { valid: true, error: null };
}

export function checkConnectivity(islands, bridges) {
  if (islands.length === 0) return true;

  const adj = new Map();
  for (const isl of islands) adj.set(isl.id, []);

  for (const [key, count] of bridges.entries()) {
    if (!count) continue;
    const { x1, y1, x2, y2 } = parseBridgeKey(key);
    const a = islands.find((i) => i.x === x1 && i.y === y1);
    const b = islands.find((i) => i.x === x2 && i.y === y2);
    if (!a || !b) continue;
    adj.get(a.id).push(b.id);
    adj.get(b.id).push(a.id);
  }

  const start = islands[0].id;
  const seen = new Set([start]);
  const stack = [start];
  while (stack.length) {
    const cur = stack.pop();
    for (const n of adj.get(cur) || []) {
      if (!seen.has(n)) {
        seen.add(n);
        stack.push(n);
      }
    }
  }

  return islands.every((i) => seen.has(i.id));
}

export function isVictory(islands, bridges) {
  // counts satisfied
  const counts = computeCurrentCounts(islands, bridges);
  for (const isl of islands) {
    if ((counts.get(isl.id) ?? 0) !== isl.value) return false;
  }

  // no crossings
  for (const [key, count] of bridges.entries()) {
    if (!count) continue;
    const { x1, y1, x2, y2 } = parseBridgeKey(key);
    const a = islands.find((i) => i.x === x1 && i.y === y1);
    const b = islands.find((i) => i.x === x2 && i.y === y2);
    if (!a || !b) continue;
    if (bridgesWouldCross(islands, bridges, a, b)) return false;
  }

  // connected
  return checkConnectivity(islands, bridges);
}
