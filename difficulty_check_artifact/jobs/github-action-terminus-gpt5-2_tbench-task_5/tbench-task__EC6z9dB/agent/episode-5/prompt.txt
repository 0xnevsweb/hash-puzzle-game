export const checkConnectivity = (islands, bridges) => {
  if (!islands.length) return true;

  const adj = new Map();
  for (const isl of islands) adj.set(isl.id, new Set());

  for (const [k, n] of bridges.entries()) {
    if (!n) continue;
    const [x1, y1, x2, y2] = k.split(',').map(Number);
    const a = islands.find((i) => i.x === x1 && i.y === y1);
    const b = islands.find((i) => i.x === x2 && i.y === y2);
    if (!a || !b) continue;
    adj.get(a.id).add(b.id);
    adj.get(b.id).add(a.id);
  }

  const start = islands[0].id;
  const q = [start];
  const seen = new Set([start]);
  while (q.length) {
    const cur = q.shift();
    for (const nxt of adj.get(cur) ?? []) {
      if (seen.has(nxt)) continue;
      seen.add(nxt);
      q.push(nxt);
    }
  }

  return seen.size === islands.length;
};

export const isVictory = (islands, bridges) => {
  const entries = [...bridges.entries()].filter(([, n]) => n);

  // defensive crossing check
  for (let i = 0; i < entries.length; i++) {
    const [k1] = entries[i];
    const [a1x, a1y, a2x, a2y] = k1.split(',').map(Number);
    const a1 = { x: a1x, y: a1y };
    const a2 = { x: a2x, y: a2y };
    for (let j = i + 1; j < entries.length; j++) {
      const [k2] = entries[j];
      const [b1x, b1y, b2x, b2y] = k2.split(',').map(Number);
      const b1 = { x: b1x, y: b1y };
      const b2 = { x: b2x, y: b2y };

      const sharesEndpoint =
        (a1.x === b1.x && a1.y === b1.y) ||
        (a1.x === b2.x && a1.y === b2.y) ||
        (a2.x === b1.x && a2.y === b1.y) ||
        (a2.x === b2.x && a2.y === b2.y);
      if (sharesEndpoint) continue;

      if (segmentCrosses(a1, a2, b1, b2)) return false;
    }
  }

  const counts = computeCurrentCounts(islands, bridges);
  for (const isl of islands) {
    if ((counts.get(isl.id) ?? 0) !== isl.value) return false;
  }

  return checkConnectivity(islands, bridges);
};
