const keyFromCoords = (a, b) => {
  // a,b are {x,y} in grid units
  const first = a.x < b.x || (a.x === b.x && a.y <= b.y) ? a : b;
  const second = first === a ? b : a;
  return `${first.x},${first.y},${second.x},${second.y}`;
};

export const makeBridgeKey = (from, to) => keyFromCoords(from, to);

export const computeCurrentCounts = (islands, bridges) => {
  const counts = new Map();
  for (const isl of islands) counts.set(isl.id, 0);

  for (const [k, n] of bridges.entries()) {
    if (!n) continue;
    const [x1, y1, x2, y2] = k.split(',').map(Number);
    const a = islands.find((i) => i.x === x1 && i.y === y1);
    const b = islands.find((i) => i.x === x2 && i.y === y2);
    if (!a || !b) continue;
    counts.set(a.id, (counts.get(a.id) ?? 0) + n);
    counts.set(b.id, (counts.get(b.id) ?? 0) + n);
  }
  return counts;
};

const segmentCrosses = (a1, a2, b1, b2) => {
  // only consider strict crossings between orthogonal segments (no shared endpoints)
  const aVert = a1.x === a2.x;
  const bVert = b1.x === b2.x;
  if (aVert === bVert) return false;

  const v1 = aVert ? a1 : b1;
  const v2 = aVert ? a2 : b2;
  const h1 = aVert ? b1 : a1;
  const h2 = aVert ? b2 : a2;

  const vx = v1.x;
  const hy = h1.y;

  const vMinY = Math.min(v1.y, v2.y);
  const vMaxY = Math.max(v1.y, v2.y);
  const hMinX = Math.min(h1.x, h2.x);
  const hMaxX = Math.max(h1.x, h2.x);

  // strict interior intersection (not at endpoints)
  if (vx <= hMinX || vx >= hMaxX) return false;
  if (hy <= vMinY || hy >= vMaxY) return false;

  return true;
};

const bridgesWouldCross = (islands, bridges, from, to, keyToIgnore = null) => {

export const canAddBridge = (islands, bridges, from, to, delta) => {
  if (!from || !to) return { valid: false, error: 'Invalid islands' };
  if (from.id === to.id) return { valid: false, error: 'Cannot connect island to itself' };

  if (!areAdjacent(islands, from, to)) {
    return { valid: false, error: 'Cannot connect non-adjacent islands' };
  }

  const key = makeBridgeKey(from, to);
  const current = bridges.get(key) ?? 0;
  const next = current + delta;

  if (next > 2) return { valid: false, error: 'Max 2 bridges per pair' };
  if (next < 0) return { valid: false, error: 'Invalid bridge count' };

  // crossing check only when adding (not when removing)
  if (delta > 0) {
    if (bridgesWouldCross(islands, bridges, from, to, key)) {
      return { valid: false, error: 'Bridges would cross' };
    }
  }

  // island capacity check
  const counts = computeCurrentCounts(islands, bridges);
  const fromCount = counts.get(from.id) ?? 0;
  const toCount = counts.get(to.id) ?? 0;

  if (delta > 0) {
    if (fromCount + delta > from.value || toCount + delta > to.value) {
      return { valid: false, error: 'Island would exceed its number' };
    }
  }

  return { valid: true, error: null };
};
