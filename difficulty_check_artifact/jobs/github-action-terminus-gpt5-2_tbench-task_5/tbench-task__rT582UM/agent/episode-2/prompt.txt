function segmentIntersectsStrict(a1, a2, b1, b2) {
  // Only considering axis-aligned segments.
  const aVert = a1.x === a2.x;
  const bVert = b1.x === b2.x;
  if (aVert === bVert) return false; // parallel do not cross in Hashi (overlap not possible with adjacency constraints)

  const vert = aVert ? { p1: a1, p2: a2 } : { p1: b1, p2: b2 };
  const horz = aVert ? { p1: b1, p2: b2 } : { p1: a1, p2: a2 };

  const vx = vert.p1.x;
  const vy1 = Math.min(vert.p1.y, vert.p2.y);

  return crosses;
}

function bridgesWouldCross(islands, bridges, from, to) {
  const a1 = { x: from.x, y: from.y };
  const a2 = { x: to.x, y: to.y };

  for (const [k, count] of bridges.entries()) {
    if (!count) continue;
    const [x1, y1, x2, y2] = k.split(',').map(Number);
    const b1 = { x: x1, y: y1 };
    const b2 = { x: x2, y: y2 };

    // If sharing endpoints, it's not a crossing.
    const sharesEndpoint =
      (a1.x === b1.x && a1.y === b1.y) ||
      (a1.x === b2.x && a1.y === b2.y) ||
      (a2.x === b1.x && a2.y === b1.y) ||
      (a2.x === b2.x && a2.y === b2.y);
    if (sharesEndpoint) continue;

    if (segmentIntersectsStrict(a1, a2, b1, b2)) return true;
  }
  return false;
}

export function computeCurrentCounts(islands, bridges) {
  const { byXY } = buildOccupancy(islands);
  const counts = new Map();
  for (const isl of islands) counts.set(isl.id, 0);

  for (const [k, n] of bridges.entries()) {
    if (!n) continue;
    const [x1, y1, x2, y2] = k.split(',').map(Number);
    const a = byXY.get(`${x1},${y1}`);
    const b = byXY.get(`${x2},${y2}`);
    if (!a || !b) continue;
    counts.set(a.id, (counts.get(a.id) ?? 0) + n);

}

export function checkConnectivity(islands, bridges) {
  if (islands.length === 0) return true;
