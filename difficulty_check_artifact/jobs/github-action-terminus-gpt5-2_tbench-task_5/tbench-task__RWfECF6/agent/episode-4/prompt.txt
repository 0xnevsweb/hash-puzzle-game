export function isVictory(islands, bridges) {
  // counts satisfied
  const counts = computeCurrentCounts(islands, bridges);
  for (const isl of islands) {
    if ((counts.get(isl.id) ?? 0) !== isl.value) return false;
  }

  // no crossings
  const keys = Array.from(bridges.entries()).filter(([, n]) => n > 0).map(([k]) => k);
  for (let i = 0; i < keys.length; i++) {
    for (let j = i + 1; j < keys.length; j++) {
      if (segmentCrosses(keys[i], keys[j])) return false;
    }
  }

  // connected
  return checkConnectivity(islands, bridges);
}
