// Hashi (Bridges) core logic utilities

export const ERR_NON_ADJACENT = 'Cannot connect non-adjacent islands';
export const ERR_MAX_PAIR = 'Max 2 bridges per pair';
export const ERR_EXCEED = 'Island would exceed its number';
export const ERR_CROSS = 'Bridges would cross';

export function canonicalBridgeKey(from, to) {
  const a = { x: from.x, y: from.y };
  const b = { x: to.x, y: to.y };
  const first = a.x < b.x || (a.x === b.x && a.y <= b.y) ? a : b;
  const second = first === a ? b : a;
  return `${first.x},${first.y},${second.x},${second.y}`;
}

function parseKey(key) {
  const [x1, y1, x2, y2] = key.split(',').map((n) => Number(n));
  return { x1, y1, x2, y2 };
}

function islandById(islands, id) {
  return islands.find((i) => i.id === id);
}

function between(v, a, b) {
  return v > Math.min(a, b) && v < Math.max(a, b);
}

export function areAdjacent(islands, from, to) {
  if (!from || !to) return false;
  if (from.id === to.id) return false;

  const sameRow = from.y === to.y;
  const sameCol = from.x === to.x;
  if (!sameRow && !sameCol) return false;

  // Must have no island strictly between them on that row/col
  if (sameRow) {
    const y = from.y;
    const blocking = islands.some((i) => i.y === y && between(i.x, from.x, to.x));
    return !blocking;
  }

  const x = from.x;
  const blocking = islands.some((i) => i.x === x && between(i.y, from.y, to.y));
  return !blocking;
}

export function computeCurrentCounts(islands, bridges) {
  const counts = new Map();
  for (const isl of islands) counts.set(isl.id, 0);

  for (const [key, n] of bridges.entries()) {
    if (!n) continue;
    const { x1, y1, x2, y2 } = parseKey(key);
    const a = islands.find((i) => i.x === x1 && i.y === y1);
    const b = islands.find((i) => i.x === x2 && i.y === y2);
    if (!a || !b) continue;
    counts.set(a.id, (counts.get(a.id) ?? 0) + n);
    counts.set(b.id, (counts.get(b.id) ?? 0) + n);
  }

  return counts;
}

function bridgesWouldCross(islands, bridges, candidateKey) {
  const c = parseKey(candidateKey);
  const cHorizontal = c.y1 === c.y2;

  const cMinX = Math.min(c.x1, c.x2);
  const cMaxX = Math.max(c.x1, c.x2);
  const cMinY = Math.min(c.y1, c.y2);
  const cMaxY = Math.max(c.y1, c.y2);

  for (const [key, n] of bridges.entries()) {
    if (!n) continue;
    if (key === candidateKey) continue;

    const b = parseKey(key);
    const bHorizontal = b.y1 === b.y2;

    // Parallel bridges don't cross; also sharing endpoints is fine.
    if (cHorizontal === bHorizontal) continue;

    const bMinX = Math.min(b.x1, b.x2);
    const bMaxX = Math.max(b.x1, b.x2);
    const bMinY = Math.min(b.y1, b.y2);
    const bMaxY = Math.max(b.y1, b.y2);

    // Ensure we're checking strict interior intersection (not endpoints)
    if (cHorizontal) {
      const ix = b.x1; // vertical x
      const iy = c.y1; // horizontal y

      const intersects =
        ix > cMinX &&
        ix < cMaxX &&
        iy > bMinY &&
        iy < bMaxY;

      if (!intersects) continue;

      // If intersection happens on an island coordinate (shouldn't due to strict), ignore.
      return true;
    } else {
      const ix = c.x1;
      const iy = b.y1;
      const intersects =
        ix > bMinX &&
        ix < bMaxX &&
        iy > cMinY &&
        iy < cMaxY;
      if (intersects) return true;
    }
  }

  return false;
}

export function canAddBridge(islands, bridges, from, to, delta) {
  // delta: +1, -1, or 0 (we'll validate add intent)
  if (!areAdjacent(islands, from, to)) {
    return { valid: false, error: ERR_NON_ADJACENT };
  }

  const key = canonicalBridgeKey(from, to);
  const current = bridges.get(key) ?? 0;
  const next = current + delta;

  if (next < 0) return { valid: true, error: null }; // removing is always ok

  if (next > 2) {
    return { valid: false, error: ERR_MAX_PAIR };
  }

  if (delta > 0) {
    // crossing check only needed if bridge will exist
    if (next > 0 && bridgesWouldCross(islands, bridges, key)) {
      return { valid: false, error: ERR_CROSS };
    }

    const counts = computeCurrentCounts(islands, bridges);
    const fromCount = counts.get(from.id) ?? 0;
    const toCount = counts.get(to.id) ?? 0;

    if (fromCount + 1 > from.value || toCount + 1 > to.value) {
      return { valid: false, error: ERR_EXCEED };
    }
  }

  return { valid: true, error: null };
}

export function checkConnectivity(islands, bridges) {
  if (!islands.length) return true;

  const adj = new Map();
  for (const isl of islands) adj.set(isl.id, []);

  for (const [key, n] of bridges.entries()) {
    if (!n) continue;
    const { x1, y1, x2, y2 } = parseKey(key);
    const a = islands.find((i) => i.x === x1 && i.y === y1);
    const b = islands.find((i) => i.x === x2 && i.y === y2);
    if (!a || !b) continue;
    adj.get(a.id).push(b.id);
    adj.get(b.id).push(a.id);
  }

  const startId = islands[0].id;
  const seen = new Set([startId]);
  const q = [startId];

  while (q.length) {
    const cur = q.shift();
    for (const nxt of adj.get(cur) ?? []) {
      if (!seen.has(nxt)) {
        seen.add(nxt);
        q.push(nxt);
      }
    }
  }

  return seen.size === islands.length;
}

export function isVictory(islands, bridges) {
  // No crossings should exist
  for (const [key, n] of bridges.entries()) {
    if (!n) continue;
    if (bridgesWouldCross(islands, bridges, key)) return false;
  }

  const counts = computeCurrentCounts(islands, bridges);
  for (const isl of islands) {
    if ((counts.get(isl.id) ?? 0) !== isl.value) return false;
  }

  return checkConnectivity(islands, bridges);
}
