// Bridge key: canonical "x1,y1,x2,y2" where endpoint with smaller (x, then y) sorts first
export function getBridgeKey(island1, island2) {
  const a = { x: island1.x, y: island1.y };
  const b = { x: island2.x, y: island2.y };
  if (a.x < b.x || (a.x === b.x && a.y < b.y)) {
    return `${a.x},${a.y},${b.x},${b.y}`;
  }
  return `${b.x},${b.y},${a.x},${a.y}`;
}
