// Bridge key: canonical "x1,y1,x2,y2" where first endpoint sorts smaller (x then y)
export function bridgeKey(a, b) {
  // a and b are {x, y} objects (grid coords)
  if (a.x < b.x || (a.x === b.x && a.y < b.y)) {
    return `${a.x},${a.y},${b.x},${b.y}`;
  }
  return `${b.x},${b.y},${a.x},${a.y}`;
}

// Parse a bridge key back into two points
export function parseBridgeKey(key) {
  const [x1, y1, x2, y2] = key.split(',').map(Number);
  return [{ x: x1, y: y1 }, { x: x2, y: y2 }];
}
