// Hashi (Bridges) logic utilities

function canonicalKey(a, b) {
  // a,b are {x,y} in grid units
  const aFirst = a.x < b.x || (a.x === b.x && a.y <= b.y);
  const p1 = aFirst ? a : b;
  const p2 = aFirst ? b : a;
  return `${p1.x},${p1.y},${p2.x},${p2.y}`;
}

export function makeBridgeKey(from, to) {
  return canonicalKey({ x: from.x, y: from.y }, { x: to.x, y: to.y });
}

function isStraight(from, to) {
  return from.x === to.x || from.y === to.y;
}

function bridgeOrientation(from, to) {
  if (from.x === to.x && from.y !== to.y) return 'V';
  if (from.y === to.y && from.x !== to.x) return 'H';
  return null;
}

function isIslandBetween(island, from, to) {
  if (from.x === to.x) {
    if (island.x !== from.x) return false;
    const minY = Math.min(from.y, to.y);
    const maxY = Math.max(from.y, to.y);
    return island.y > minY && island.y < maxY;
  }
  if (from.y === to.y) {
    if (island.y !== from.y) return false;
    const minX = Math.min(from.x, to.x);
    const maxX = Math.max(from.x, to.x);
    return island.x > minX && island.x < maxX;
  }
  return false;
}

export function areAdjacent(islands, from, to) {
  if (!isStraight(from, to)) return false;
  // Adjacent means no island in between on same row/col
  for (const isl of islands) {
    if (isl.id === from.id || isl.id === to.id) continue;
    if (isIslandBetween(isl, from, to)) return false;
  }
  return true;
}

function segmentsCross(a1, a2, b1, b2) {
  // Only consider proper crossings (not sharing endpoints). Grid axis-aligned.
  const oa = bridgeOrientation(a1, a2);
  const ob = bridgeOrientation(b1, b2);
  if (!oa || !ob) return false;
  if (oa === ob) return false;

  const h = oa === 'H' ? { p1: a1, p2: a2 } : { p1: b1, p2: b2 };
  const v = oa === 'V' ? { p1: a1, p2: a2 } : { p1: b1, p2: b2 };

  const hx1 = Math.min(h.p1.x, h.p2.x);
  const hx2 = Math.max(h.p1.x, h.p2.x);

  const vy2 = Math.max(v.p1.y, v.p2.y);
