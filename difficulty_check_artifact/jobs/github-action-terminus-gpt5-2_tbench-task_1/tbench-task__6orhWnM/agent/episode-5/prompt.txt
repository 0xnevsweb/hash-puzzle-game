// Hashi (Bridges) game logic utilities

export const ERROR_NON_ADJACENT = 'Cannot connect non-adjacent islands';
export const ERROR_MAX_2 = 'Max 2 bridges per pair';
export const ERROR_EXCEED = 'Island would exceed its number';
export const ERROR_CROSS = 'Bridges would cross';

export function canonicalBridgeKey(a, b) {
  // a,b are islands with grid coordinates
  const p1 = { x: a.x, y: a.y };
  const p2 = { x: b.x, y: b.y };
  const first = p1.x < p2.x || (p1.x === p2.x && p1.y <= p2.y) ? p1 : p2;
  const second = first === p1 ? p2 : p1;
  return `${first.x},${first.y},${second.x},${second.y}`;
}

function buildCoordMap(islands) {
  const m = new Map();
  for (const isl of islands) m.set(`${isl.x},${isl.y}`, isl);
  return m;
}

export function areAdjacent(islands, a, b) {
  if (!a || !b) return false;
  if (a.id === b.id) return false;
  if (a.x !== b.x && a.y !== b.y) return false;

  const coordMap = buildCoordMap(islands);
  if (a.x === b.x) {
    const x = a.x;
    const [y1, y2] = a.y < b.y ? [a.y, b.y] : [b.y, a.y];
    for (let y = y1 + 1; y <= y2 - 1; y++) {
      if (coordMap.has(`${x},${y}`)) return false;
    }
    return true;
  }

  const y = a.y;
  const [x1, x2] = a.x < b.x ? [a.x, b.x] : [b.x, a.x];
  for (let x = x1 + 1; x <= x2 - 1; x++) {
    if (coordMap.has(`${x},${y}`)) return false;
  }
  return true;
}

function parseKey(key) {
  const [x1, y1, x2, y2] = key.split(',').map((n) => Number(n));
  return { x1, y1, x2, y2 };
}

function isHorizontal(seg) {
  return seg.y1 === seg.y2;
}

function segmentsCross(segA, segB) {
  // Only perpendicular crossings count; shared endpoints are not crossing.
  const aH = isHorizontal(segA);
  const bH = isHorizontal(segB);
  if (aH === bH) return false;

  const h = aH ? segA : segB;
  const v = aH ? segB : segA;

  const hx1 = Math.min(h.x1, h.x2);
  const hx2 = Math.max(h.x1, h.x2);
  const hy = h.y1;

  const vy1 = Math.min(v.y1, v.y2);
  const vy2 = Math.max(v.y1, v.y2);
  const vx = v.x1;

  // strict interior intersection only (not at endpoints)
  const intersects = vx > hx1 && vx < hx2 && hy > vy1 && hy < vy2;
  return intersects;
}

export function bridgesCross(bridges) {
  const segs = [];
  for (const [key, count] of bridges.entries()) {
    if (!count) continue;
    segs.push(parseKey(key));
  }
  for (let i = 0; i < segs.length; i++) {
    for (let j = i + 1; j < segs.length; j++) {
      if (segmentsCross(segs[i], segs[j])) return true;
    }
  }
  return false;
}

export function computeCurrentCounts(islands, bridges) {
  const counts = new Map();
  for (const isl of islands) counts.set(isl.id, 0);

  for (const [key, n] of bridges.entries()) {
    if (!n) continue;
    const { x1, y1, x2, y2 } = parseKey(key);
    const a = islands.find((i) => i.x === x1 && i.y === y1);
    const b = islands.find((i) => i.x === x2 && i.y === y2);
    if (!a || !b) continue;
    counts.set(a.id, (counts.get(a.id) ?? 0) + n);
    counts.set(b.id, (counts.get(b.id) ?? 0) + n);
  }
  return counts;
}

export function checkConnectivity(islands, bridges) {
  if (islands.length === 0) return true;

  const adj = new Map();
  for (const isl of islands) adj.set(isl.id, []);

  for (const [key, n] of bridges.entries()) {
    if (!n) continue;
    const { x1, y1, x2, y2 } = parseKey(key);
    const a = islands.find((i) => i.x === x1 && i.y === y1);
    const b = islands.find((i) => i.x === x2 && i.y === y2);
    if (!a || !b) continue;
    // any positive bridge count implies connectivity edge
    adj.get(a.id).push(b.id);
    adj.get(b.id).push(a.id);
  }

  const start = islands[0].id;
  const seen = new Set([start]);
  const q = [start];
  while (q.length) {
    const cur = q.shift();
    for (const nxt of adj.get(cur) || []) {
      if (!seen.has(nxt)) {
        seen.add(nxt);
        q.push(nxt);
      }
    }
  }

  return seen.size === islands.length;
}
