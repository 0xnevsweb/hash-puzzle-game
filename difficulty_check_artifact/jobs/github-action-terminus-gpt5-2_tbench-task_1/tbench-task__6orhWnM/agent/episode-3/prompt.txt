export function bridgesCross(bridges) {
  const segs = [];
  for (const [key, count] of bridges.entries()) {
    if (!count) continue;
    segs.push(parseKey(key));
  }
  for (let i = 0; i < segs.length; i++) {
    for (let j = i + 1; j < segs.length; j++) {
      if (segmentsCross(segs[i], segs[j])) return true;
    }
  }
  return false;
}

export function computeCurrentCounts(islands, bridges) {
  const counts = new Map();
  for (const isl of islands) counts.set(isl.id, 0);

  for (const [key, n] of bridges.entries()) {
    if (!n) continue;
    const { x1, y1, x2, y2 } = parseKey(key);
    const a = islands.find((i) => i.x === x1 && i.y === y1);
    const b = islands.find((i) => i.x === x2 && i.y === y2);
    if (!a || !b) continue;
    counts.set(a.id, (counts.get(a.id) ?? 0) + n);
    counts.set(b.id, (counts.get(b.id) ?? 0) + n);
  }
  return counts;
}
